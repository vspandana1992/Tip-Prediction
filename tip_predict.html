class data_cleaning(object):
    def __init__(self,Data):
        self.Data      = Data
    '''Clean up the data points by the following:
        [a] Trip distance cannot be negative
        [b] Tips cannot be negative
        [c] Total amount cannot be negative and has to be a minimum of 2.5
        [d] Fare amount cannot be negative and has to be a minimum of 2.5
        
    '''
    def _remove_nonsensical_data(self):
        filtered_data = self.Data[(self.Data.total_amount>2.5) & (self.Data.fare_amount>2.5) & (self.Data.tip_amount>=0.) & (self.Data.trip_distance>0.) & (~self.Data.RatecodeID==99)].reset_index(drop=True)
        return filtered_data
    ''' from scipy.stats import pearsonr
        pearsonr(Data2.trip_distance,Data2.tip_amount) = 0.77
        The total fare depends on the trip distance and total fare amount so look for outliers in the trip distance data points and remove them, since the data is not
        gaussian distributed use interquartile range to filter out outliers.'''
    def _remove_outliers(self):
        stats            = self.Data.trip_distance.describe()
        iqr              = stats['75%']-stats['25%']
        lower,upper      = stats['25%']-1.5*iqr , stats['75%']+1.5*iqr
        data_wo_outliers1 = self.Data[(self.Data.trip_distance > lower) & (self.Data.trip_distance < upper)].reset_index(drop=True)
        stats            = data_wo_outliers1.total_amount.describe()
        iqr              = stats['75%']-stats['25%']
        lower,upper      = stats['25%']-1.5*iqr , stats['75%']+1.5*iqr
        data_wo_outliers2 = data_wo_outliers1[(data_wo_outliers1.total_amount > lower) & (data_wo_outliers1.total_amount < upper)].reset_index(drop=True)
        return data_wo_outliers2

class feature_extraction(object):
    """
    This function create new variables based on present variables in the dataset Data. It creates:
    . Week: int {1,2,3,4,5}, Week a transaction was done
    . Week_day: int [0-6], day of the week a transaction was done
    . Month_day: int [0-30], day of the month a transaction was done
    . Hour: int [0-23], hour the day a transaction was done
    . Shift type: int {1=(7am to 3pm), 2=(3pm to 11pm) and 3=(11pm to 7am)}, shift of the day  
    . Speed_mph: float, speed of the trip
    . Trip duration : float , duration of the trip
    . With_tip: int {0,1}, 1 = transaction with tip, 0 transction without tip
    """
    # derive time variables
    def _time_features(self,Data):
        ref_week               = datetime.datetime(2017,1,1).isocalendar()[1] 
        Data['Week']      = Data.tpep_pickup_datetime.apply(lambda x:x.isocalendar()[1])-ref_week+1
        Data['Week_day']  = Data.tpep_pickup_datetime.apply(lambda x:x.isocalendar()[2])
        Data['Month_day'] = Data.tpep_pickup_datetime.apply(lambda x:x.day)
        Data['Hour']      = Data.pickup

        # create 8 hour time shift variable:  1=(7am to 3pm), 2=(3pm to 11pm) and 3=(11pm to 7am)
        Data['Shift_type'] = 3
        Data.loc[Data[(Data.Hour>=7) & (Data.Hour<15)].index,'Shift_type'] = 1
        Data.loc[Data[(Data.Hour>=15) & (Data.Hour<23)].index,'Shift_type'] = 2
        
         # Trip duration 
        Data['Trip_duration'] = ((Data.tpep_dropoff_datetime -Data.tpep_pickup_datetime ).apply(lambda x:x.total_seconds()/60.))
        return Data
   
    
    # create variable for Speed
    def _speed_feature(self,Data):
        Data['Speed_mph'] = Data.trip_distance/(Data.Trip_duration/60)
        # ignore speed > 200 and nans
        Data = Data[~((Data.Speed_mph.isnull()) | (Data.Speed_mph>200))]
        return Data
    def _tip_feature(self,Data):
        # create with_tip variable
        Data['With_tip'] = (Data.tip_percentage>0)*1
        return Data

from sklearn import metrics
from sklearn.model_selection import cross_val_score,GridSearchCV

# define a function that help to train models and perform cv
def optimized_model(algorithm,train,predictors,target,scoring_method,performCV=True,printFeatureImportance=True,cv_folds=5):
    # train the algorithm on data
    algorithm.fit(train[predictors],train[target])
    #predict on train set:
    train_predictions = algorithm.predict(train[predictors])
    if scoring_method == 'roc_auc':
        train_predprob = algorithm.predict_proba(train[predictors])[:,1]

    #perform cross-validation
    if performCV:
        cv_score = cross_val_score(algorithm,train[predictors],train[target],cv=cv_folds,scoring=scoring_method)
        #print model report
        print('Model report:::')
        if scoring_method == 'roc_auc':
            print("Accuracy :: {0}".format(metrics.accuracy_score(train[target].values,train_predictions)))
            print("AUC Score for Training data :: {0}".format(metrics.roc_auc_score(train[target], train_predprob)))
        if scoring_method == 'neg_mean_squared_error':
            print("Accuracy:: {0}".format(metrics.mean_squared_error(train[target].values,train_predictions)))
    if performCV:
        print("CV Score - Mean : %.7g | Std : %.7g | Min : %.7g | Max : %.7g" % (np.mean(cv_score),np.std(cv_score),np.min(cv_score),np.max(cv_score)))
    #print feature importance
    if printFeatureImportance:
        feat_imp = PD.Series(algorithm.feature_importances_,predictors).sort_values(ascending=False)
        feat_imp.plot(kind='bar',title='Feature Importances')
        PLT.ylabel('Feature Importance Score')
        PLT.show()

# optimize n_estimator through grid search
def optimize_num_trees(algorithm,param_test,scoring_method,train,predictors,target):
    grid_search = GridSearchCV(estimator=algorithm, param_grid = param_test, scoring=scoring_method,n_jobs=2,iid=False,cv=5)
    grid_search.fit(train[predictors],train[target])
    return grid_search

# plot optimization results
def plot_opt_results(alg):

    plt.plot(cv_results[1],cv_results[0])
    plt.xlabel('# trees')
    plt.ylabel('score')
    plt.title('optimization report')
